318template<typename Lambda>
319void UArticyFlowPlayer::ShadowedOperation(Lambda Operation) const
320{
321 if (std::is_null_pointer<Lambda>::value)
322 return;
323 if (!ensure(GetGVs()))
324 {
325 UE_LOG(LogArticyRuntime, Warning, TEXT("FlowPlayer cannot get GVs!"));
326 return;
327 }
328 if (!ensure(ShadowLevel < ShadowLevelLimit))
329 {
330 UE_LOG(LogArticyRuntime, Warning, TEXT("More than 100 nested ShadowedOperations, probably there's an infinite loop in the Flow!"));
331 return;
332 }
333
334 //push shadow state
335 ++ShadowLevel;
336
337 //notify on push
338 GetGVs()->PushState(ShadowLevel);
339 GetGVs()->PushSeen();
340 UArticyDatabase::Get(this)->PushState(ShadowLevel);
341 OnShadowOpStart.Broadcast();
342
343 //execute the operation
344 Operation();
345
346 //notify on pop
347 OnShadowOpEnd.Broadcast();
348 UArticyDatabase::Get(this)->PopState(ShadowLevel);
349 GetGVs()->PopSeen();
350 GetGVs()->PopState(ShadowLevel);
351
352 //pop shadow state
353 if (ensure(ShadowLevel > 0))
354 --ShadowLevel;
355}
